#include <fcntl.h> //open,creat
#include <sys/types.h> //open
#include <sys/stat.h>
#include <errno.h> //perror, errno

- errno változó  -  benne van a hiba száma (ha nem nulla akkor hiba volt/van)
- access(filename,F_OK);  -  létezik-e ilyen nevû fájl
- creat(filename,permissions);  -  csinál ilyen nevû fájlt
- close(f);  -  fájl bezárása

<MINTAKÓD "fájl megnyitása + hibakezelés">
    int f;
    f = open(argv[1],O_RDONLY);
    f = open(argv[2],O_WRONLY|O_CREAT|O_TRUNC,S_IRUSR|S_IWUSR);
        O_TRUNC  -  ha már létezett a fájl, törli a tatrtalmát
        O_CREAT  -  ha még nem létezett, hozza létre
        S_IRUSR|S_IWUSR  -  jogosultságok (S_IRUSR == reading by the owner)
    if (f<0)
    {
        perror("Error at opening the file\n");
        exit(1);
    }
</MINTAKÓD>
