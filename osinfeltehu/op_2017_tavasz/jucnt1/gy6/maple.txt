fibonacci2:=proc(m)
 first:=0;
 sec:=1;
 l:=true;
 c:=0;
 kov:=0;
 L:=[first mod m,sec mod m];
 db:=1;
 i:=0;
 while l do
   
  if c<=1 then
   kov:=c;
   
  else
  
   kov:=first+sec;
   first:=sec;
   sec:=kov;
   if first mod m=0 and sec mod m=1 then
    l:=false;
   
   else
     L:=[op(L),sec mod m];
     db:=db+1;
   end if;
  end if;
  c:=c+1;
 end do;
 L:=subsop(1=NULL,L);
 return [db,L];
end;
